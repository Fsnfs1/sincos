"GODLIKE!" : "NICE!", "#10B981"); if (numFilled && denFilled) { gameState = "SUCCESS"; document.getElementById("instruction").innerText = "✨ 建构完成！ ✨"; document.getElementById("geometry-pane").classList.add("main-success"); updateUI(); setTimeout(() => { document.getElementById("geometry-pane").classList.remove("main-success"); if (gameState !== "GAMEOVER") nextMission(); // 阻滞状态竞争 }, 700); } else { updateInstruction(); } updateUI(); } else { flashEdge = { id: edgeType, color: "#EF4444", timer: Date.now() + 400 }; drawGeometry(); handlePenalty(); } } function updateInstruction() { if(gameState === "DRAWING_PHASE") { document.getElementById("instruction").innerText = "🔥 难度升级：请从发光顶点向下滑动，画出高线"; document.getElementById("instruction").style.color = "#EF4444"; } else { // 动态视觉脚手架：依据填写状态提供单一明确的视觉焦点 if (!numFilled) { document.getElementById("instruction").innerText = `🎯 第一步：点击图形中的 [${stepNames[currentMission.steps.num]}] 作为分子`; activateSlot("numerator-slot"); document.getElementById("denominator-slot").classList.remove("active"); } else if (!denFilled) { document.getElementById("instruction").innerText = `🎯 第二步：点击图形中的 [${stepNames[currentMission.steps.den]}] 作为分母`; document.getElementById("numerator-slot").classList.remove("active"); activateSlot("denominator-slot"); } document.getElementById("instruction").style.color = "#FACC15"; } }
